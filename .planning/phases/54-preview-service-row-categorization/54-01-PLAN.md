D-12 bucketing waterfall (first match wins, per row):
  1. Blank PSN → ERROR/BLANK_PSN_ID
  2. Blank team code → ERROR/BLANK_TEAM_CODE
  3. Unknown team short code (teamRepository.findByShortName empty) → ERROR/UNKNOWN_TEAM_CODE
  4. PSN already seen in this tab → ERROR/DUPLICATE_IN_TAB
  5. MatchType.FUZZY → FUZZY_SUGGESTION
  6. MatchType.EXACT:
     - suggestedSeasonId == null OR findBySeasonIdAndDriverId empty → NEW_ASSIGNMENT
     - findBySeasonIdAndDriverId present + same team id → UNCHANGED
     - findBySeasonIdAndDriverId present + different team id → CONFLICT
  7. MatchType.NONE → NEW_DRIVER

    public DriverSheetImportPreview preview(String sheetUrl) throws IOException {
        String spreadsheetId = googleSheetsService.extractSpreadsheetId(sheetUrl);
        List<String> allTabs = googleSheetsService.getSheetNames(spreadsheetId);

        List<String> yearTabs = allTabs.stream()
                .filter(name -> YEAR_TAB_PATTERN.matcher(name).matches())
                .sorted()   // 4-digit numeric strings sort == ascending year
                .toList();

        log.info("Driver sheet preview: sheet {} has {} total tabs, {} match year pattern",
                spreadsheetId, allTabs.size(), yearTabs.size());

        List<TabPreview> tabPreviews = new ArrayList<>();
        for (String tab : yearTabs) {
            int year = Integer.parseInt(tab);
            List<List<Object>> rows = googleSheetsService.readRangeFromSheet(spreadsheetId, tab, "A:C");
            tabPreviews.add(categorizeTab(tab, year, rows));
        }

        return new DriverSheetImportPreview(List.copyOf(tabPreviews));
    }

    private TabPreview categorizeTab(String tabName, int year, List<List<Object>> rows) {
        // Season auto-match (D-02, D-03)
        List<Season> seasonCandidates = seasonRepository.findByYear(year);
        UUID suggestedSeasonId = null;
        String ambiguousReason = null;
        if (seasonCandidates.size() == 1) {
            suggestedSeasonId = seasonCandidates.get(0).getId();
        } else if (seasonCandidates.isEmpty()) {
            ambiguousReason = "No season found for year " + year;
        } else {
            ambiguousReason = "Multiple seasons for year " + year;
        }

        List<NewDriverRow> newDrivers = new ArrayList<>();
        List<NewAssignmentRow> newAssignments = new ArrayList<>();
        List<ConflictRow> conflicts = new ArrayList<>();
        List<FuzzySuggestionRow> fuzzySuggestions = new ArrayList<>();
        List<UnchangedRow> unchanged = new ArrayList<>();
        List<ErrorRow> errors = new ArrayList<>();
        Set<String> seenPsnsInTab = new HashSet<>();

        if (rows.size() > 1) {
            for (int i = 1; i < rows.size(); i++) {
                List<Object> row = rows.get(i);
                String rawPsn      = row.size() > 0 ? cellToString(row.get(0)) : "";
                String rawTeamCode = row.size() > 2 ? cellToString(row.get(2)) : "";
                String psn         = rawPsn.trim();
                String teamCode    = rawTeamCode.trim();

                // D-12 waterfall — first match wins
                if (psn.isBlank()) {
                    errors.add(new ErrorRow(rawPsn, rawTeamCode, ErrorReason.BLANK_PSN_ID));
                    continue;
                }
                if (teamCode.isBlank()) {
                    errors.add(new ErrorRow(rawPsn, rawTeamCode, ErrorReason.BLANK_TEAM_CODE));
                    continue;
                }
                Optional<Team> teamOpt = teamRepository.findByShortName(teamCode);
                if (teamOpt.isEmpty()) {
                    errors.add(new ErrorRow(rawPsn, rawTeamCode, ErrorReason.UNKNOWN_TEAM_CODE));
                    continue;
                }
                if (!seenPsnsInTab.add(psn)) {
                    errors.add(new ErrorRow(rawPsn, rawTeamCode, ErrorReason.DUPLICATE_IN_TAB));
                    continue;
                }

                Team sheetTeam = teamOpt.get();
                DriverMatchingService.MatchResult match = driverMatchingService.findDriver(psn);

                if (match.type() == DriverMatchingService.MatchType.FUZZY) {
                    Driver candidate = match.driver();
                    fuzzySuggestions.add(new FuzzySuggestionRow(
                            psn,
                            candidate.getId(),
                            candidate.getPsnId(),
                            candidate.getNickname(),
                            match.similarity(),
                            teamCode));
                } else if (match.type() == DriverMatchingService.MatchType.EXACT) {
                    Driver matched = match.driver();
                    Optional<SeasonDriver> existingSd = (suggestedSeasonId == null)
                            ? Optional.empty()
                            : seasonDriverRepository.findBySeasonIdAndDriverId(suggestedSeasonId, matched.getId());
                    if (existingSd.isEmpty()) {
                        newAssignments.add(new NewAssignmentRow(psn, matched.getId(), teamCode));
                    } else {
                        SeasonDriver sd = existingSd.get();
                        if (sd.getTeam().getId().equals(sheetTeam.getId())) {
                            unchanged.add(new UnchangedRow(psn, matched.getId(), teamCode));
                        } else {
                            conflicts.add(new ConflictRow(
                                    psn,
                                    matched.getId(),
                                    sd.getId(),
                                    sd.getTeam().getShortName(),
                                    teamCode));
                        }
                    }
                } else { // MatchType.NONE
                    newDrivers.add(new NewDriverRow(psn, teamCode));
                }
            }
        }

        log.info("Tab {} categorized: {} new, {} assignments, {} conflicts, {} fuzzy, {} unchanged, {} errors",
                tabName, newDrivers.size(), newAssignments.size(), conflicts.size(),
                fuzzySuggestions.size(), unchanged.size(), errors.size());

        return new TabPreview(
                tabName, year, suggestedSeasonId, ambiguousReason,
                List.copyOf(newDrivers), List.copyOf(newAssignments),
                List.copyOf(conflicts), List.copyOf(fuzzySuggestions),
                List.copyOf(unchanged), List.copyOf(errors));
    }

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-54-01 | Tampering / Information Disclosure (SSRF vector via crafted sheet URL) | `DriverSheetImportService.preview(sheetUrl)` delegating to `GoogleSheetsService.extractSpreadsheetId(url)` | accept (transferred) | Phase 54 does **not** perform any new HTTP I/O; all fetch goes through the existing `GoogleSheetsService`, which is already the canonical, credential-scoped Sheets client used by Phase 23+ import flows. `extractSpreadsheetId` is a regex parse on the user-supplied URL, not a request. Any SSRF risk lives in `GoogleSheetsService` and is out of scope. Rationale: introducing URL allow-listing in Phase 54 would duplicate policy that belongs at the infrastructure layer. |
| T-54-02 | Injection / Stored XSS risk (cell payload flows into DB strings in Phase 55) | Sheet row cells → `NewDriverRow.psnId` / `teamShortName` / `ErrorRow.rawPsnId` | mitigate (deferred to Phase 55) | Phase 54 stores **no** data and renders **no** HTML. Mitigation for persistence/rendering belongs in Phase 55: the controller writes values through JPA (parameterized queries — no SQL injection surface) and Thymeleaf auto-escapes by default.
