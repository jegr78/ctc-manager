```java
@Transactional
public Season removeTeamFromSeason(UUID seasonId, UUID teamId) {
    var season = findById(seasonId);
    var team = teamRepository.findById(teamId)
            .orElseThrow(() -> new EntityNotFoundException("Team not found: " + teamId));

    // D-25 strict guard: if any PhaseTeam (across ALL phases of this season) references the team, refuse
    var allPhases = seasonPhaseService.findAllPhases(seasonId);
    for (var phase : allPhases) {
        var pt = phaseTeamRepository.findByPhaseIdAndTeamId(phase.getId(), teamId);
        if (pt.isPresent()) {
            throw new BusinessRuleException(
                    "Cannot remove team from season: team is still assigned to phase '"
                    + (phase.getLabel() != null && !phase.getLabel().isBlank() ? phase.getLabel() : phase.getPhaseType()) + "'. Remove it from phases first.");
        }
    }
    season.removeTeam(team);
    log.info("Removed team {} from season {}", teamId, seasonId);
    return seasonRepository.save(season);
}
```
