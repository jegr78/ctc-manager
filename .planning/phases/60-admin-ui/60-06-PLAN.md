```java
@PostMapping("/{id}/swiss/generate")
public String generateSwissRound(@PathVariable UUID id, RedirectAttributes redirectAttributes) {
    var regular = seasonPhaseService.findRegularPhase(id);
    try {
        swissPairingService.generateNextRound(regular.getId());
        redirectAttributes.addFlashAttribute("successMessage", "Next Swiss round generated");
    } catch (BusinessRuleException | IllegalStateException e) {
        log.warn("Cannot generate Swiss round: {}", e.getMessage());
        redirectAttributes.addFlashAttribute("errorMessage", e.getMessage());
    }
    return "redirect:/admin/seasons/" + id + "/swiss";
}
```

```java
@PostMapping("/{id}/generate")
public String generate(@PathVariable UUID id,
                       @RequestParam int numberOfRounds,

                       RedirectAttributes redirectAttributes) {

        log.warn("Cannot generate matchdays: {}", e.getMessage());

    }
    return "redirect:/admin/seasons/" + id;
}
```
