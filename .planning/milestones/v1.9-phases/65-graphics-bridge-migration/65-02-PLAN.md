private Map<UUID, Integer> calculateBuchholzScores(UUID seasonId) {
    var season = seasonRepository.findById(seasonId).orElse(null);
    if (season == null) return Map.of();
    Map<UUID, UUID> successionMap = season.buildSuccessionMap();
    var standings = calculateStandings(seasonId);   // <-- bridge call (line 209)
    Map<UUID, Integer> pointsMap = standings.stream()
            .collect(Collectors.toMap(s -> s.getTeam().getId(), TeamStanding::getPoints));
    List<Race> races = raceRepository.findByMatchdaySeasonIdAndPlayoffMatchupIsNull(seasonId);
    Map<UUID, Set<UUID>> opponents = new HashMap<>();

        if (race.isBye() || race.getAwayTeam() == null) continue;
        UUID home = successionMap.getOrDefault(race.getHomeTeam().getId(), race.getHomeTeam().getId());
        UUID away = successionMap.getOrDefault(race.getAwayTeam().getId(), race.getAwayTeam().getId());
        opponents.computeIfAbsent(home, k -> new HashSet<>()).add(away);
        opponents.computeIfAbsent(away, k -> new HashSet<>()).add(home);

        buchholz.put(entry.getKey(), sum);
    }
    return buchholz;
}

        * Calculates Buchholz scores for the given phase. Buchholz is the sum of opponents' points
        * accumulated across the entire season (including any prior phases) — opponent aggregation
        * is intentionally season-scoped because the race finder operates on season FK.
        *
        * <p>For LEAGUE / per-group GROUPS the season-level calculation produces values equivalent
        * to a phase-only calculation because the phase's matchdays are the only matchdays. For
        * GROUPS combined-view, Buchholz is display-only (NOT used as a tiebreaker).
        */
       private Map<UUID, Integer> calculateBuchholzScoresForPhase(SeasonPhase phase) {
           // Pitfall 5: phase.getSeason() is always non-null — NOT NULL FK in schema, eager-loaded under OSIV.
           // The previous null-guard via seasonRepository.findById(seasonId).orElse(null) is dropped.
           var season = phase.getSeason();
           Map<UUID, UUID> successionMap = season.buildSuccessionMap();

           // D-04b: use canonical phase-aware API instead of the seasonId bridge.
           var standings = calculateStandings(phase.getId(), null);
           Map<UUID, Integer> pointsMap = standings.stream()
                   .collect(Collectors.toMap(s -> s.getTeam().getId(), TeamStanding::getPoints));

           // raceRepository finder STAYS season-scoped (Buchholz spans the whole season's opponents).
           List<Race> races = raceRepository.findByMatchdaySeasonIdAndPlayoffMatchupIsNull(season.getId());
           Map<UUID, Set<UUID>> opponents = new HashMap<>();
           for (Race race : races) {
               if (race.isBye() || race.getAwayTeam() == null) continue;

           }

                       .mapToInt(oppId -> pointsMap.getOrDefault(oppId, 0))

           }

           return buchholz;
       }
